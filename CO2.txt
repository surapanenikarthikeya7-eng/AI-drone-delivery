#include <iostream>
#include <vector>
using namespace std;

/* ==========================
   B-TREE (Simplified)
   ========================== */

class BTree {
    vector<int> keys;

public:
    void insert(int key) {
        keys.push_back(key);
    }

    bool search(int key) {
        for (int x : keys)
            if (x == key)
                return true;
        return false;
    }
};

/* ==========================
   B+ TREE (Simplified)
   ========================== */

class BPlusTree {
    vector<int> records;

public:
    void insert(int key) {
        records.push_back(key);
    }

    bool search(int key) {
        for (int x : records)
            if (x == key)
                return true;
        return false;
    }
};

/* ==========================
   SEGMENT TREE
   ========================== */

class SegmentTree {
    vector<int> tree;
    int n;

    void build(vector<int>& arr, int node, int start, int end) {
        if(start == end)
            tree[node] = arr[start];
        else {
            int mid = (start + end) / 2;

            build(arr, 2 * node, start, mid);
            build(arr, 2 * node + 1, mid + 1, end);

            tree[node] =
                tree[2 * node] +
                tree[2 * node + 1];
        }
    }

    int query(int node, int start,
              int end, int l, int r) {

        if(r < start || end < l)
            return 0;

        if(l <= start && end <= r)
            return tree[node];

        int mid = (start + end) / 2;

        return query(2 * node, start, mid, l, r) +
               query(2 * node + 1, mid + 1, end, l, r);
    }

public:
    SegmentTree(vector<int>& arr) {
        n = arr.size();
        tree.resize(4 * n);
        build(arr, 1, 0, n - 1);
    }

    int rangeSum(int l, int r) {
        return query(1, 0, n - 1, l, r);
    }
};

/* ==========================
   FENWICK TREE
   ========================== */

class FenwickTree {
    vector<int> bit;
    int n;

public:
    FenwickTree(int size) {
        n = size;
        bit.assign(n + 1, 0);
    }

    void update(int index, int value) {
        while(index <= n) {
            bit[index] += value;
            index += index & (-index);
        }
    }

    int prefixSum(int index) {
        int sum = 0;

        while(index > 0) {
            sum += bit[index];
            index -= index & (-index);
        }

        return sum;
    }

    int rangeSum(int l, int r) {
        return prefixSum(r) -
               prefixSum(l - 1);
    }
};

int main() {

    cout << "MULTI-WAY TREE ANALYSIS PROJECT\n\n";

    // B-Tree
    BTree bt;
    bt.insert(10);
    bt.insert(20);
    bt.insert(30);

    cout << "B-Tree Search 20: ";
    cout << (bt.search(20) ? "Found" : "Not Found") << endl;

    // B+ Tree
    BPlusTree bpt;
    bpt.insert(100);
    bpt.insert(200);
    bpt.insert(300);

    cout << "B+ Tree Search 200: ";
    cout << (bpt.search(200) ? "Found" : "Not Found") << endl;

    // Segment Tree
    vector<int> arr = {1,2,3,4,5};

    SegmentTree st(arr);

    cout << "Segment Tree Range Sum (1,3): ";
    cout << st.rangeSum(1,3) << endl;

    // Fenwick Tree
    FenwickTree ft(5);

    for(int i=0;i<5;i++)
        ft.update(i+1, arr[i]);

    cout << "Fenwick Tree Range Sum (2,5): ";
    cout << ft.rangeSum(2,5) << endl;

    return 0;
}
